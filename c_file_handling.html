why files?
Without files
All data lives in RAM. When the program ends or the computer turns off, everything is gone.
With files
Data is saved to disk permanently. Other programs can read it. Data survives after the program exits.
Use cases
Student records, game saves, logs, configuration files, reports, data export.

In C, a file is represented by a FILE* pointer (file pointer). fopen() returns this pointer — it's how every other function knows which file to work with. Always check if fopen() returned NULL (file not found / permission denied) before using it.
#include <stdio.h> FILE* fp — file pointer always check fp != NULL always call fclose(fp)

FILE *fp;

// "w" — create or overwrite
fp = fopen("data.txt", "w");

// "r" — read existing
fp = fopen("data.txt", "r");

// "a" — add to end of file
fp = fopen("log.txt", "a");

// ALWAYS check!
if (fp == NULL) {
  printf("Error opening file!\n");
  return 1;
}
Common mistake with "w" mode: If the file already exists, "w" immediately erases all its contents before writing. If you want to keep existing data and add more, use "a" (append) instead.
"r" — read only "w" — write (erases!) "a" — append (safe) NULL = file open failed
write_file.c
#include <stdio.h>

int main() {
  FILE *fp;
  fp = fopen("students.txt", "w");
  if (fp == NULL) {
    printf("Cannot open file!\n");
    return 1;
  }
  fprintf(fp, "Alice 95\n");  // like printf but to file
  fprintf(fp, "Bob   82\n");
  fprintf(fp, "Carol 78\n");
  fputs("Done writing!\n", fp); // string to file
  fclose(fp);  // flush + close
  printf("File written successfully!\n");
  return 0;
}

read_file.c
#include <stdio.h>

int main() {
  FILE *fp;
  char line[100];
  fp = fopen("students.txt", "r");
  if (fp == NULL) {
    printf("File not found!\n");
    return 1;
  }
  // Read line by line until EOF
  while (fgets(line, 100, fp) != NULL) {
    printf("%s", line);  // print each line
  }
  fclose(fp);
  return 0;
}

student_records.c — complete example
#include <stdio.h>

/* Write student records to a file */
void writeStudents() {
  FILE *fp = fopen("students.txt", "w");
  if (fp == NULL) { printf("Error!\n"); return; }

  char names[][20] = {"Alice", "Bob", "Carol"};
  int  marks[]     = {95, 82, 78};

  for (int i = 0; i < 3; i++)
    fprintf(fp, "%s %d\n", names[i], marks[i]);

  fclose(fp);
  printf("Records written!\n");
}

/* Read and display student records */
void readStudents() {
  FILE *fp = fopen("students.txt", "r");
  if (fp == NULL) { printf("File not found!\n"); return; }

  char name[20];
  int  mark;

  printf("Name      Mark\n");
  printf("----------\n");

  while (fscanf(fp, "%s %d", name, &mark) == 2) {
    printf("%-9s %d\n", name, mark);
  }
  fclose(fp);
}

/* Append a new student */
void appendStudent(char *name, int mark) {
  FILE *fp = fopen("students.txt", "a");
  if (fp == NULL) { printf("Error!\n"); return; }
  fprintf(fp, "%s %d\n", name, mark);
  fclose(fp);
  printf("Student added!\n");
}

int main() {
  writeStudents();
  readStudents();
  appendStudent("David", 88);
  readStudents();  // read again — David appears
  return 0;
}
