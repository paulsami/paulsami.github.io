// Nested loops: outer = row (i), inner = col (j)
for (int i = 1; i <= 5; i++) {         // outer loop
    for (int j = 1; j <= 5; j++) {     // inner loop
        printf("%4d", i * j);             // product
    }
    printf("\n");                          // new row
}
