menu_loop.c
int choice;
for (;;) {             // infinite loop
  printf("\\n=== MENU ===\\n");
  printf("1. Greet\\n");
  printf("2. Count\\n");
  printf("3. Joke\\n");
  printf("4. Quit\\n");
  scanf("%d", &choice);
  if (choice == 4) break;
  if (choice == 1)
    printf("Hello, World!\\n");
  else if (choice == 2)
    printf("1 2 3 4 5\\n");
  else if (choice == 3)
    printf("Why do C programmers..\\n");
  else
    printf("Invalid choice!\\n");
}                      // loop back
printf("Goodbye!\\n");

menu_switch.c — cleaner with switch
int choice;
for (;;) {
  printf("\\n=== MENU ===\\n");
  printf("1. Greet  2. Count\\n");
  printf("3. Joke   4. Quit\\n");
  scanf("%d", &choice);

  switch (choice) {
    case 1:
      printf("Hello, World!\\n");
      break;                // break exits switch
    case 2:
      printf("1 2 3 4 5\\n");
      break;
    case 3:
      printf("Why do C programmers\\n");
      printf("mix up Halloween\\n");
      printf("and Christmas?\\n");
      printf("Oct 31 == Dec 25!\\n");
      break;
    case 4:
      printf("Goodbye!\\n");
      goto done;            // or use a flag
    default:
      printf("Invalid choice!\\n");
  }
}
done:;

/* Alternative: use a flag variable */
int running = 1;
while (running) {
  /* ... same menu ... */
  if (choice == 4) running = 0; // no break needed
}
Three patterns for exiting a menu loop:

1. if + break — simplest, fine for small menus.
2. switch + goto done — break inside switch only exits the switch, not the loop, so goto or a flag is needed.
3. Flag variable — while (running) with running = 0 to exit. Avoids goto entirely and is very readable.
if + break — simplest switch — needs goto or flag flag variable — most readable
