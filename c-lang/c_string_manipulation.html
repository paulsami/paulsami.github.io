void reverse(char s[]) {
  int l=0, r=strlen(s)-1;
  while (l < r) {
    char tmp = s[l];
    s[l] = s[r];
    s[r] = tmp;
    l++; r--;
  }
}
// "MANGO" → "OGNAM"

palindrome.c
int isPalindrome(char s[]) {
  int l=0, r=strlen(s)-1;
  while (l < r) {
    if (s[l] != s[r])
      return 0; // not palindrome
    l++; r--;
  }
  return 1;   // is palindrome
}
// "RADAR" → 1 (yes!)
// "HELLO" → 0 (no)

word_count.c
int wordCount(char s[]) {
  int count=0, inWord=0;
  for (int i=0; s[i]!='\0'; i++) {
    if (s[i]==' ' || s[i]=='\t') {
      inWord = 0;  // leaving word
    } else if (!inWord) {
      count++;     // entering new word
      inWord = 1;
    }
  }
  return count;
}
// "I love C" → 3 words

// Replace all occurrences of a char
void replaceChar(char s[], char old, char new) {
  for (int i=0; s[i]!='\0'; i++) {
    if (s[i] == old)
      s[i] = new;  // replace in place
  }
}

// Find position of first char
int findChar(char s[], char c) {
  for (int i=0; s[i]!='\0'; i++)
    if (s[i]==c) return i; // found!
  return -1;  // not found
}
