break_search.c
int arr[] = {4, 7, 2, 9, 1};
int target = 9, found = -1;
for (int i = 0; i < 5; i++) {
    if (arr[i] == target) {
        found = i;
        break; // stop searching
    }
}
printf(found >= 0 ? "found at %d\n" : "not found\n", found);

continue_evens.c
// Print only even numbers 0–9
for (int i = 0; i < 10; i++) {
    if (i % 2 != 0) {
        continue; // skip odd numbers
