array_loop.c
int a[6] = {4, 7, 2, 9, 1, 5};
int n = sizeof(a) / sizeof(a[0]); // = 6
int sum = 0, max = a[0];

for (int i = 0; i < n; i++) {
    sum += a[i];           // accumulate sum
    if (a[i] > max) max = a[i]; // track max
}
printf("Sum=%d Max=%d Avg=%.1f\n",
        sum, max, (float)sum/n);

pitfalls.c — common C array mistakes
int a[5] = {10,20,30,40,50};

// ❌ BUG 1: Off-by-one — reads garbage/crashes
int x = a[5]; // valid indices: 0-4 only!

// ❌ BUG 2: Wrong loop bound
for (int i = 0; i <= 5; i++) // should be i < 5
    printf("%d", a[i]); // a[5] is out of bounds

// ❌ BUG 3: sizeof on pointer (in function)
void bad(int arr[]) {
    int n = sizeof(arr)/sizeof(arr[0]); // WRONG!
}   // arr decays to pointer → sizeof = 8

// ✅ FIX: pass size explicitly
void good(int arr[], int n) { /* use n */ }
